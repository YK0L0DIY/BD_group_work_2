create table Modelo(
	Modelo text Primary Key,
	nLugares integer,
	Autonomia integer	
);

create table Aviao(
	NomeA text Primary Key,
	Marca text
);

create table modelo_de(
	Modelo text References Modelo on delete Cascade,
	NomeA text References Aviao on delete Cascade,
	Primary Key(Modelo,NomeA)
);

create table Funcionario(
	nBI integer Primary Key,
	NIF integer,
	NomeF text,
	DataN date
);

create table Piloto(
	nBI integer References Funcionario on delete Cascade,
	nCarta integer,
	anoInicio integer,
	Primary Key(nBI)
);

create table Admin(
	nBI integer References Funcionario on delete Cascade,
	Primary Key(nBI)
);

create table Hosp(
	nBI integer References Funcionario on delete Cascade,
	altura integer,
	peso integer,
	anoInicio integer,
	Primary Key(nBI)
);

create table Comiss(
	nBI integer References Funcionario on delete Cascade,
	altura integer,
	peso integer,
	anoInicio integer,
	Primary Key(nBI)
);

create table Voo_planeado(
	nVoo integer Primary Key,
	destino text,
	origem text,
	diaSemana text,
	horasPartida time,
	horasChegada time,
	kms integer
);

create table Voo_executado(
	nVooExecutado integer,
	nVoo integer References Voo_planeado on delete Cascade,
	nPassageiros integer,
	horaExChegada time,
	horaExPartida time,
	dataVoo date,
	Primary Key(nVooExecutado, nvoo)

);

create table AUsado(
	nVoo integer,
	nVooExecutado integer ,
	NomeA text,
	primary key(nvoo,nvooexecutado,nomeA)
	
);

create table PilotoUsado(
	nVoo integer ,
	nVooExecutado integer ,
	nBI integer,
	primary key(nvoo,nvooexecutado,nbi)
);

create table HospUsado(
	nVoo integer,
	nVooExecutado integer ,
	nBI integer,
	primary key(nvoo,nvooexecutado,nbi)
);

create table ComissUsado(
	nVoo integer ,
	nVooExecutado integer ,
	nBI integer,
	primary key(nvoo,nvooexecutado,nbi)
);

create table RestricaoPilotos(
	AnosExpMin integer,
	nPilotos integer,
	nVoo integer references Voo_planeado on delete Cascade,
	Primary Key(nvoo,AnosExpMin,nPilotos)
	
);

create table RestricaoHosp(
	PesoMax integer,
	nHosp integer,
	nVoo integer references Voo_planeado on delete Cascade,
	Primary Key(nvoo,PesoMax,nHosp)
);

create table RestricaoComiss(
	AltMin integer,
	nComiss integer,
	nVoo integer references Voo_planeado on delete Cascade,
	Primary Key(nvoo,AltMin,nComiss)
);
