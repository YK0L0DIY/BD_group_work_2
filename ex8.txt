create table Modelo(
	Modelo text Primary Key,
	nLugares integer,
	Autonomia integer	
);

create table Aviao(
	NomeA text Primary Key,
	Marca text
);

create table modelo_de(
	Modelo References Modelo on delete Cascade,
	NomeA References Aviao on delete Cascade,
	Primary Key(Modelo,NomeA)
);

create table Funcionarios(
	nBI integer Primary Key,
	NIF integer,
	NomeF text,
	DataN date
);

create table Piloto(
	nBI References Funcionario on delete Cascade,
	nCarta integer,
	anosInicio integer,
	Primary Key(nBI)
);

create table Admin(
	nBI References Funcionario on delete Cascade,
	Primary Key(nBI)
);

create table Hosp(
	nBI References Funcionario on delete Cascade,
	altura integer,
	peso integer,
	anosInicio integer,
	Primary Key(nBI)
);

create table Comiss(
	nBI References Funcionario on delete Cascade,
	altura integer,
	peso integer,
	anosInicio integer,
	Primary Key(nBI)
);

create table Voo_executado(
	nVooExecutado integer Primary Key,
	nVoo References on delete Cascade Voo_planeado
);

create table Voo_planeado(
	nVoo integer Primary Key,
	destino text,
	origem text,
	diaSemana text,
	horasPartida time,
	horasChegada time,
	kms integer
);

create table AUsado(
	nVoo references Voo_planeado on delete Cascade,
	nVooExecutado References Voo_executado on delete Cascade,
	NomeA References Aviao on delete Cascade
);

create table PilotoUsado(
	nVoo references Voo_planeado on delete Cascade,
	nVooExecutado References Voo_executado on delete Cascade,
	nBI References Funcionario on delete Cascade
);

create table HospUsado(
	nVoo references Voo_planeado on delete Cascade,
	nVooExecutado References Voo_executado on delete Cascade,
	nBI References Funcionario on delete Cascade
);

create table ComissUsado(
	nVoo references Voo_planeado on delete Cascade,
	nVooExecutado References Voo_executado on delete Cascade,
	nBI References Funcionario on delete Cascade
);

create table RestricaoPilotos(
	AnosExpMin integer,
	nPilotos integer,
	Primary Key(AnosExpMin,nPilotos)
);

create table RestricaoHosp(
	PesoMax integer,
	nHosp integer,
	Primary Key(PesoMax,nPilotos)
);

create table RestricaoComiss(
	AltMin integer,
	nComiss integer,
	Primary Key(AltMin,nPilotos)
);

create table PrecisaPilotos(
	nVoo References Voo_planeado on delete Cascade,
	AnosExpMin References ResticaoPiloto on delete Cascade,
	nPilotos References RestricaoPiloto on delete Cascade,
	Primary Key(nVoo, AnosExpMin, nPilotos)
);

create table PrecisaHosp(
	nVoo References Voo_planeado on delete Cascade,
	PesoMax References ResticaoHosp on delete Cascade,
	nHosp References RestricaoHosp on delete Cascade,
	Primary Key(nVoo, PesoMax, nHosp)
);

create table PrecisaComiss(
	nVoo References Voo_planeado on delete Cascade,
	AltMin References ResticaoComiss on delete Cascade,
	nComiss References RestricaoComiss on delete Cascade,
	Primary Key(nVoo, AltMin, nComiss)
);
